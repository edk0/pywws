    "Convert pressure from hectopascals/millibar to inches of mercury"
    if hPa is None:

[docs]def temp_f(c):
    "Convert temperature from Celsius to Fahrenheit"
    if c is None:
        return None
    return (c * 9.0 / 5.0) + 32.0
